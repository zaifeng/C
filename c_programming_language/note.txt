#include <stdio.h>   // <> 与 ""的区别是 ""包含的是用户自己的，<>包含的是系统的lib库文件

int main()           //C语言都是由函数和变量组成，其中必须包含一个main入口函数
{
    printf("hello world\n");    //该函数来自stdio.h严格来说不是C语言的一部分
    return 0 ;                  //C语言原则上应该返回一个值，表明代码执行成功或者失败，在shell环境下可以用$?来获取这个返回值
}  //方法体用花括号 括起来

几个转义字符

\t      制表符
\a      响铃
\b      回退符
\n      换行符
\r      回车符
\'      单引号
\"      双引号
\\      反斜杠
